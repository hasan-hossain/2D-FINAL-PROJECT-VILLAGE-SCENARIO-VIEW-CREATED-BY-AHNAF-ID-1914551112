void handleMouse(int button, int state, int x, int y) {

if (button == GLUT_LEFT_BUTTON)
	{
	    speed += 0.05f;
    }


else if (button == GLUT_RIGHT_BUTTON)
	{
	    speed -= 0.05f;
	}
glutPostRedisplay();
}
void handleKeypress(unsigned char key, int x, int y) {
	switch (key) {
case 's'://stops
    speed = 0.0f;
    break;
case 'r'://runs
    speed = 0.02f;
    break;
glutPostRedisplay();
	}}
